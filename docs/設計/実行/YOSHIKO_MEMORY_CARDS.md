## アニメーション仕様

### Phase 1: 扇状展開（自動）
```
初期状態: 1枚のカード（裏向き・中央配置）
トリガー: ScrollTrigger（セクション到達時・top 70%）

GSAP Timeline:
  1. 1枚のカードが8枚に分裂（stagger: 0.1s）
  2. 各カードが扇状に回転配置
     rotateZ: (i - 3.5) * 8°（-28° 〜 +28°）
     x: (i - 3.5) * 65px
     y: |i - 3.5| * -15px（扇のアーチ）
  3. 各カードに年代色のグラデーションが現れる
  4. duration: 0.8s, ease: back.out(1.4)
```

### Phase 2: カードめくり（ユーザー操作）
```
トリガー: クリック / タップ

アニメーション:
  1. クリックされたカードのz-indexが最前面に
  2. rotateY: 0° → 180°（0.6s, ease: power2.inOut）
  3. 裏面が表示:
     上部: イラスト写真（カード幅100%×高さ70px）
     中央左: 日付テキスト（0.65rem）
     中央右: 年齢バッジ（カード色の丸角ピル）
     下部: 出来事テキスト（0.65rem, line-height 1.5）
  4. VOICEVOX音声が自動再生
  
戻し: カード外クリックでactiveCard = null
      （裏面は表示されたまま。一度めくったカードは裏のまま）
```

### Phase 3: 全8枚めくり完了（ゲーミフィケーション）
```
判定: flipped.size === 8

演出:
  1. 全カードが少し縮小（scale: 0.85）
  2. 中央にヨシコのアバター出現（fadeIn + scale 0→1）
  3. 吹き出し:「ね、同じ時代を歩いてきたでしょう？」
  4. VOICEVOX音声再生（yoshiko-memory-complete.mp3）
  5. CTAボタン:「ヨシコに自分の話もしてみる」
     → onChatbotOpen() → AIChatWidget が開く
```

  const containerRef = useRef<HTMLDivElement>(null);
  const allFlipped = flipped.size === 8;

  useEffect(() => {
    const initGsap = async () => {
      try {
        const gsap = (await import('gsap')).default;
        const { ScrollTrigger } = await import('gsap/ScrollTrigger');
        gsap.registerPlugin(ScrollTrigger);
        if (!containerRef.current) return;

        const cards = containerRef.current.querySelectorAll('.memory-card');
        gsap.set(cards, { opacity: 0, scale: 0.3, rotateZ: 0, x: 0, y: 0 });

        ScrollTrigger.create({
          trigger: containerRef.current,
          start: 'top 70%',
          onEnter: () => {
            gsap.to(cards, {
              opacity: 1, scale: 1,
              rotateZ: (i: number) => (i - 3.5) * 8,
              x: (i: number) => (i - 3.5) * 65,
              y: (i: number) => Math.abs(i - 3.5) * -15,
              stagger: 0.1,
              duration: 0.8,
              ease: 'back.out(1.4)',
            });
          },
        });
      } catch {
        if (containerRef.current) {
          containerRef.current.querySelectorAll('.memory-card')
            .forEach(el => ((el as HTMLElement).style.opacity = '1'));
        }
      }
    };
    initGsap();
  }, []);

  const handleFlip = (id: number) => {
    setActiveCard(id);
    setFlipped(prev => new Set(prev).add(id));
    const card = CARDS.find(c => c.id === id);
    if (card?.voice) new Audio(card.voice).play().catch(() => {});
    window.dataLayer?.push({ event: 'memory_card_flip', card_id: id, card_label: card?.label });
  };

  useEffect(() => {
    if (allFlipped) {
      new Audio('/audio/yoshiko-memory-complete.mp3').play().catch(() => {});
      window.dataLayer?.push({ event: 'memory_cards_all_flipped' });
    }
  }, [allFlipped]);

  return (
    <Box ref={containerRef} sx={{ position: 'relative', minHeight: 420, my: 6 }}>
      <Typography variant="h6" sx={{ textAlign: 'center', mb: 1, fontWeight: 700 }}>
        ヨシコの思い出
      </Typography>
      <Typography variant="body2" sx={{ textAlign: 'center', mb: 4, color: 'text.secondary' }}>
        カードをめくると、ヨシコの67年間が見えます
      </Typography>

      <Box sx={{
        display: 'flex', justifyContent: 'center', alignItems: 'center',
        position: 'relative', height: 220, perspective: '1200px',
      }}>
        {CARDS.map((card) => (
          <Box key={card.id} className="memory-card"
            onClick={() => handleFlip(card.id)}
            sx={{
              position: 'absolute',
              width: 130, height: 185, cursor: 'pointer',
              transformStyle: 'preserve-3d',
              transition: 'transform 0.6s cubic-bezier(0.4, 0, 0.2, 1)',
              transform: flipped.has(card.id) ? 'rotateY(180deg)' : 'rotateY(0deg)',
              zIndex: activeCard === card.id ? 10 : 1,
            }}>
            {/* 表面 */}
            <Box sx={{
              position: 'absolute', inset: 0, backfaceVisibility: 'hidden',
              background: `linear-gradient(135deg, ${card.color}, ${card.color}dd)`,
              borderRadius: 2.5, p: 2,
              display: 'flex', flexDirection: 'column', alignItems: 'center',
              justifyContent: 'center', boxShadow: 3,
            }}>
              <Typography sx={{ fontSize: '2.5rem', mb: 1 }}>{card.icon}</Typography>
              <Typography variant="caption" sx={{ fontWeight: 700, fontSize: '0.75rem' }}>
                {card.era}
              </Typography>
              <Typography variant="caption" sx={{ fontSize: '0.7rem', mt: 0.5 }}>
                {card.label}
              </Typography>
            </Box>
            {/* 裏面 */}
            <Box sx={{
              position: 'absolute', inset: 0, backfaceVisibility: 'hidden',
              transform: 'rotateY(180deg)',
              background: '#FFFDF7', borderRadius: 2.5, p: 1.5,
              display: 'flex', flexDirection: 'column',
              boxShadow: '0 4px 20px rgba(0,0,0,0.15)',
              border: `2px solid ${card.color}`,
              overflow: 'hidden',
            }}>
              <Box component="img" src={card.photo} alt={card.label}
                sx={{ width: '100%', height: 70, objectFit: 'cover', borderRadius: 1.5, mb: 1 }} />
              <Box sx={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', mb: 0.5 }}>
                <Typography variant="caption" sx={{ fontWeight: 700, color: 'text.secondary', fontSize: '0.6rem' }}>
                  {card.date}
                </Typography>
                <Box sx={{ background: card.color, borderRadius: 10, px: 1, py: 0.2 }}>
                  <Typography variant="caption" sx={{ fontWeight: 700, fontSize: '0.6rem' }}>
                    {card.age}
                  </Typography>
                </Box>
              </Box>
              <Typography variant="caption" sx={{ fontSize: '0.62rem', lineHeight: 1.5, color: '#333' }}>
                {card.text}
              </Typography>
            </Box>
          </Box>
        ))}
      </Box>

      {/* 全8枚めくり完了 */}
      {allFlipped && (
        <Box sx={{ textAlign: 'center', mt: 5, animation: 'fadeInUp 0.8s ease-out' }}>
          <Typography variant="body1" sx={{ fontStyle: 'italic', mb: 2, color: 'text.secondary' }}>
            「ね、同じ時代を歩いてきたでしょう？」
          </Typography>
          <Button variant="contained" size="large" onClick={onChatbotOpen}
            sx={{ borderRadius: 8, px: 4, py: 1.5, fontSize: '1rem' }}>
            ヨシコに自分の話もしてみる
          </Button>
        </Box>
      )}
    </Box>
  );
}
